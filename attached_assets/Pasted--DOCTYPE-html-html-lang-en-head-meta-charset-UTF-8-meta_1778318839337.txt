function mf(t,p){
const s=state[t.id];
const q=(document.getElementById(‘search’).value||’’).toLowerCase();

if(!ms)return false;
if(filt===‘mandatory’)return t.mand;
if(filt===‘mustpay’)return !t.def&&(t.hi>0||t.mid>0);
if(filt===‘deferrable’)return t.def;
if(filt===‘blocked’)return s.status===‘blocked’;
if(filt===‘notdone’)return !s.done;

if(filt===‘hasnote’)return !!s.note;

if(filt===‘saving’)return !!t.saving;

}
